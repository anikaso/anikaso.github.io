#Data Set Details: 

#As of August 2004, a large Toyota car dealership offers purchasers of new Toyota cars the option to buy their used car as part of a trade-in. 
#The dealer then sells the used cars for a small profit. The `ToyotaCorolla.csv` data set includes the sales price(in EUROs) and other information on the used cars. 
#The total number of records in the dataset is 718, which is the count of Toyota cars in the inventory.

#Data Dictionary:
Variables Descriptions
Price Offer Price in EUROs
_Age_0804 Age in months as in August 2004
_MfgYear Year of Manufacture of the car
KM Accumulated Kilometers on odometer
Automatic Automatic ( (Yes=1, No=0)
Boardcomputer Boardcomputer (Yes=1, No=0)
_SportModel Sport Model (Yes=1, No=0)
Cylinders Number of cylinders
_CentralLock Central Lock (Yes=1, No=0)
_PowerSteering Power Steering (Yes=1, No=0)
Weight Weight in Kilograms
_ParkingAssistant Parking assistance system (Yes=1, No=0)
ABS Anti-Lock Brake System (Yes=1, No=0)
_FuelType Fuel Type (Petrol, Diesel, CNG)

def average_price_year(mfg_year, fuel_type, price_min, price_max, min_km, max_km):
    price_range = toyota.where('Price', are.between(price_min, price_max))
    KM_range = toyota.where('KM', are.between(min_km, max_km))
    mfg_choice = toyota.where('Mfg_Year', are.equal_to(mfg_year))
    fuel_choice = toyota.where('Fuel_Type', are.containing(fuel_type))
    
    total_count = fuel_choice.num_rows
    count = KM_range.num_rows
    
    average_price = np.average(KM_range.column('Price')) 
    percentage = (count / total_count) 

    return percentage, average_price
