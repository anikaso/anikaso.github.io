Yahoo Stock Market Predictor¶

This project uses data on the daily price high, price low, and volume of yahoo stocks sold to approximate how a dollar stock price increase correlates to the number of stocks traded. Moreover, this project uses visualizations and functions to develop correlations between stock price increase and number of stocks traded for yahoo. Lastly, this is useful to create predictions about the companies stock trades and stock value.

yahoo_data = Table.read_table('Book2.csv')

def clean_and_convert_to_numeric(column):
    '''Cleans the column by removing commas and converting to numeric type'''
    cleaned_column = []
    for value in column:
        if isinstance(value, str):
            value = value.replace(',', '')
        try:
            cleaned_column.append(float(value))
        except ValueError:
            cleaned_column.append(np.nan) 
    return cleaned_column

yahoo_data = yahoo_data.with_column('High', clean_and_convert_to_numeric(yahoo_data.column('High')))
yahoo_data = yahoo_data.with_column('Volume', clean_and_convert_to_numeric(yahoo_data.column('Volume')))
yahoo_data = yahoo_data.with_column('Low', clean_and_convert_to_numeric(yahoo_data.column('Low')))
yahoo_data = yahoo_data.with_column('Close*', clean_and_convert_to_numeric(yahoo_data.column('Close*')))

price_diff = np.array(yahoo_data.column('High')) - np.array(yahoo_data.column('Low'))
yahoo_data = yahoo_data.with_column('Stock Price Increase', price_diff)

def predict_volume(stock_price):
    '''Predicts the volume of yahoo stocks sold at a certain days stock price increase'''

    nearby = yahoo_data.where('Stock Price Increase', are.between(stock_price - 50, stock_price + 40))
    
    if len(nearby.column('Volume')) == 0:
        return np.nan 
    
    return np.mean(nearby.column('Volume'))

#USE by enetering and stock price increase into the variable stock_price

predicted_volume = predict_volume(stock_price)
